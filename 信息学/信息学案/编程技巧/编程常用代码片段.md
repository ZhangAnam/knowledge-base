## 📁 基础算法模板

### 1. 快速幂
```cpp
long long qpow(long long a, long long b) {
    long long res = 1;
    while (b) {
        if (b & 1) res = res * a;
        a = a * a;
        b >>= 1;
    }
    return res;
}
```

### 2. 归并排序
```cpp
void merge_sort(int l, int r) {
    if (l >= r) return;
    int mid = (l + r) >> 1;
    merge_sort(l, mid);
    merge_sort(mid+1, r);
    merge(l, mid, r);
}
```

### 3. 快速排序
```cpp
int partition(vector<int>& a, int l, int r) {
    int x = a[r], i = l-1;
    for (int j = l; j < r; j++)
        if (a[j] <= x) swap(a[++i], a[j]);
    swap(a[++i], a[r]);
    return i;
}
void quick_sort(int l, int r) {
    if (l >= r) return;
    int p = partition(a, l, r);
    quick_sort(l, p-1);
    quick_sort(p+1, r);
}
```

## 📁 图论模板

### 7. DFS遍历
```cpp
void dfs(int u, int fa) {
    for (int v : g[u]) {
        if (v == fa) continue;
        dfs(v, u);
    }
}
```

### 8. BFS最短路
```cpp
queue<int> q;
dist[start] = 0;
q.push(start);
while (!q.empty()) {
    int u = q.front(); q.pop();
    for (auto [v, w] : g[u]) {
        if (dist[v] > dist[u] + w) {
            dist[v] = dist[u] + w;
            q.push(v);
        }
    }
}
```

## 📁 常用技巧

### 11. 最大公约数
```cpp
int gcd(int a, int b) {
    return b ? gcd(b, a%b) : a;
}
```

### 12. 判断素数
```cpp
bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i*i <= n; i++)
        if (n % i == 0) return false;
    return true;
}
```

```cpp
int l = 0, r = n-1;
while (l < r) {
    int mid = (l+r)/2;

    else l = mid+1;

```
