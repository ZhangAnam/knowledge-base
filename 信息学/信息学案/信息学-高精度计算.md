## 📌 加法

```cpp
vector<int> add(vector<int> a, vector<int> b) {
    vector<int> c;
    int t = 0;
    for (int i = 0; i < max(a.size(), b.size()); i++) {
        if (i < a.size()) t += a[i];
        if (i < b.size()) t += b[i];
        c.push_back(t % 10);
        t /= 10;
    }
    if (t) c.push_back(t);
    return c;
}
```

## 📌 乘法

```cpp
vector<int> mul(vector<int> a, int b) {
    vector<int> c;
    long long t = 0;
    for (int i = 0; i < a.size(); i++) {
        t += a[i] * b;
        c.push_back(t % 10);
        t /= 10;
    }
    while (t) { c.push_back(t % 10); t /= 10; }
    return c;
}
```
